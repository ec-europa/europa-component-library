import Playground from '@ecl/website-components/Playground';

## Draw borders

To draw a border on an element, use one of the following classes:

<Playground>
  <div className="demo-item ecl-u-border-all">.ecl-u-border-all</div>
  <div className="demo-item ecl-u-border-top">.ecl-u-border-top</div>
  <div className="demo-item ecl-u-border-right">.ecl-u-border-right</div>
  <div className="demo-item ecl-u-border-bottom">.ecl-u-border-bottom</div>
  <div className="demo-item ecl-u-border-left">.ecl-u-border-left</div>
</Playground>

## Border width

To change the width of a border, or remove it, use one of the following classes:

<Playground>
  <div className="demo-item ecl-u-border-all ecl-u-border-width-1">
    .ecl-u-border-width-1
  </div>
  <div className="demo-item ecl-u-border-all ecl-u-border-width-2">
    .ecl-u-border-width-2
  </div>
  <div className="demo-item ecl-u-border-all ecl-u-border-width-4">
    .ecl-u-border-width-4
  </div>
  <div className="demo-item ecl-u-border-all ecl-u-border-width-8">
    .ecl-u-border-width-8
  </div>
  <div className="demo-item ecl-u-border-all ecl-u-border-width-0">
    .ecl-u-border-width-0
  </div>
</Playground>

## Border color

To change the color of a border, use the `ecl-u-border-color-*` class,
with any of <a href="https://v2--europa-component-library.netlify.com/ec/guidelines/colours/">available colors</a>.

<Playground>
  <div className="demo-item ecl-u-border-all ecl-u-border-width-2 ecl-u-border-color-black">
    .ecl-u-border-color-black
  </div>
  <div className="demo-item ecl-u-border-all ecl-u-border-width-2 ecl-u-border-color-primary">
    .ecl-u-border-color-primary
  </div>
  <div className="demo-item ecl-u-border-all ecl-u-border-width-2 ecl-u-border-color-success">
    .ecl-u-border-color-success
  </div>
  <div className="demo-item ecl-u-border-all ecl-u-border-width-2 ecl-u-border-color-warning">
    .ecl-u-border-color-warning
  </div>
  <div className="demo-item ecl-u-border-all ecl-u-border-width-2 ecl-u-border-color-grey-20">
    .ecl-u-border-color-grey-20
  </div>
</Playground>
